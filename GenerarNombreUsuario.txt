	SiNo

		Escribir "2. personaje + nombre"
		Leer option2
		Escribir "Digite el número deseado: "
		Leer number
		Si option2 = 1 Entonces
			nombre_final = Subcadena(nombre,0,3) + Subcadena(personaje_favorito,1,4)

		FinSi
